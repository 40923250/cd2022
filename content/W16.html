Y = 0.25
Z = 0

#主臂長度
a = 0.142
b = 0.1588
#c =
d1 = 0.0132
d2 = 0.0445
d3 = 0
d4 = 0.1061
#角度
#Aa = math.radians(60+5)
deg = 180/math.pi
Ab = deg*(78.38+5)
Ac = deg*(41+5)
Ad = deg*(0-5)
print(O)
print("")

print(D)
